        const findCount = (N, K, Arr) => {
            let counter = 0;

Q.4 Even Odd

        const findEvenOdd = (N, Arr) => {
            let sumeven = 0;
            let sumodd = 0;
            for (let i = 0; i < N; i++) {
                if (Arr[i] % 2 === 0) {
                    sumeven += Arr[i];
                }
                else {
                    sumodd += Arr[i];
                }
            }
            return [sumeven, sumodd];
        };

Q.9 Unique Color Shirt

        function Unique_Shirts(arr, N) {
            let count = 0;
            for (let i = 0; i < arr.length - 1; i++) {
                let repeat = false;
                for (let j = i + 1; j < arr.length; j++) {
                    if (arr[i] == arr[j]) {
                        arr[j] = 0;
                        repeat = true;
                    }
                }
                if (repeat) {
                    arr[i] = 0;
                }
            }
            for (let k = 0; k < arr.length; k++) {
                if (arr[k] > 0)
                    count = count + 1;
            }
            return count;
        }

Q.10 Min and Max

        function arrayMin(arr) {
            let len = arr.length;
            let min = Infinity;
            while (len--) {
                if (arr[len] < min) {
                    min = arr[len];
                }
            }
            return min;
        }
        function arrayMax(arr) {
            let len = arr.length;
            let max = -Infinity;
            while (len--) {
                if (arr[len] > max) {
                    max = arr[len];
                }
            }
            return max;
        }

Q.11 Birthday Game

        function Birthday_Game(arr, D, M) {
            let counter = 0;
            for (let i = 0; i < arr.length - 1; i++) {
                let subArray = arr.slice(i, i + M);
                if (subArray.length === M) {
                    let sum = subArray.reduce((a, b) => a + b, 0);
                    if (sum === D) {
                        counter += 1;
                    }
                }
            }
            return counter;
        }
